      SUBROUTINE   TSINS
     I                  (VOLUME,NUM,DELTAT,ROW,OFFSET,STKIND,
     I                   STTRAN,INMODE,A,B,MEMSB,AMDCD,GAPCD,FRC,
     M                   PRKEY)
C
C     + + + PURPOSE + + +
C     Write a primitive tsget/tsput instruction
C
C     + + + DUMMY ARGUMENTS + + +
      INTEGER    AMDCD,DELTAT,FRC,GAPCD,INMODE,MEMSB(2),NUM,
     $           OFFSET,PRKEY,ROW,STKIND,STTRAN,VOLUME
      REAL       A,B
C
C     + + + ARGUMENT DEFINITIONS + + +
C     VOLUME - ???
C     NUM    - ???
C     DELTAT - time step in minutes for source/target
C     ROW    - ???
C     OFFSET - ???
C     STKIND - ???
C     STTRAN - ???
C     INMODE - ???
C     A      - multiplying factor for the linear transform A*X + B
C     B      - shift value for the linear transform
C     MEMSB  - ???
C     AMDCD  - ???
C     GAPCD  - ???
C     FRC    - ???
C     PRKEY  - ???
C
C     + + + LOCAL VARIABLES + + +
      INTEGER    FILE,FORMT(2),NULLFG,VOLCOD,
     $           REC(15),I15
      REAL       RREC(15)
C
C     + + + EQUIVALENCES + + +
      EQUIVALENCE  (REC(1),RREC(1))
C
C     + + + INTRINSICS + + +
      INTRINSIC  ABS
C
C     + + + EXTERNALS + + +
      EXTERNAL   PUTTSI
C
C     + + + END SPECIFICATIONS + + +
C
      I15= 15
C
      IF (VOLUME .EQ. 1) THEN
C       may be tss or seq
        IF (NUM .LT. 0) THEN
C         sequential file
          FILE  = -NUM
          VOLCOD= 1
        ELSE
C         tss dataset
          FILE  = NUM
          VOLCOD= 4
        END IF
C
      ELSE IF (VOLUME .EQ. 2) THEN
C       may be wdms or dss
        IF (NUM .LT. 0) THEN
C         dss dataset
          FILE  = -NUM
          VOLCOD= 6
        ELSE
C         wdms dataset
          FILE  = NUM
          VOLCOD= -MEMSB(2)
        END IF
C
      ELSE
C       scratch pad
        VOLCOD= VOLUME
        FILE  = NUM
C
      END IF
C
      NULLFG= 0
      IF (VOLCOD .EQ. 3) THEN
C       inpad to inpad - check for null instruction
        IF (FILE .EQ. ROW) THEN
          IF (STKIND .LT. 3 .AND. STTRAN .EQ. 1) THEN
            IF (INMODE .EQ. 0) THEN
C             if (a .ne. 1.0 .or. b .ne. 0.0) go to 10
              IF ((ABS(A-1.0)) .GT. 1.0E-5 .OR. (ABS(B)) .GT. 0.0)
     $                GO TO 10
C               null instruction
                NULLFG= 1
 10           CONTINUE
            END IF
          END IF
        END IF
      END IF
C
      IF (VOLCOD .EQ. 1) THEN
C       format codes (for seq input) have been stored in member
C       subscript fields
        FORMT(1)= MEMSB(1)
        FORMT(2)= MEMSB(2)
      ELSE IF (VOLCOD .LT. 0) THEN
C       quality of data for wdms was stored in member
C       subscript field
        FORMT(1)= MEMSB(1)
        FORMT(2)= 0
      ELSE
C       format field is not meaningful
        FORMT(1)= 0
        FORMT(2)= 0
      END IF
C
      IF (NULLFG .EQ. 0) THEN
C       write primitive instruction
        PRKEY   = PRKEY+ 1
        REC(1)  = VOLCOD
        REC(2)  = FILE
        REC(3)  = DELTAT
        REC(4)  = ROW
        REC(5)  = OFFSET
        REC(6)  = STKIND
        REC(7)  = STTRAN
        REC(8)  = INMODE
        RREC(9) = A
        RREC(10)= B
        REC(11) = FORMT(1)
        REC(12) = FORMT(2)
        REC(13) = AMDCD
        REC(14) = GAPCD
        REC(15) = FRC
        CALL PUTTSI (I15,PRKEY,REC)
      END IF
C
      RETURN
      END
