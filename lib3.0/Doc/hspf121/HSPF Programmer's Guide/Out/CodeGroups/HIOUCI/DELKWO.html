      SUBROUTINE   DELKWO
     I                   (KWDTYP,KNAME)

      INTEGER      KWDTYP
      CHARACTER*12 KNAME

C     + + + ARGUMENT DEFINITIONS + + +

      INCLUDE 'cucim.inc'

      INTEGER       I,FOUND
C
C     + + + END SPECIFCATIONS + + +
C

      I = 0
      FOUND = 0
 10   CONTINUE
        I = I + 1
        IF (KNAME.EQ.UKWD(I) .AND. KWDTYP.EQ.UKWDTY(I)) THEN

        END IF
      IF (FOUND.EQ.0) GO TO 10

      END
