      SUBROUTINE SDELAY
     I                 (HUNSEC)
C
C     + + + PURPOSE + + +
C     delay specified amount of time
C
C     + + + DUMMY ARGUMENTS + + +
      INTEGER     HUNSEC
C
C     + + + ARGUMENT DEFINTIONS + + +
C     HUNSEC - hundredths of a second to delay
C
C     + + + LOCAL VARIABLES + + +
      INTEGER     TIM, OTIM
C
C     + + + END SPECIFICATIONS + + +
C
      CALL TIMER(TIM)
      OTIM = TIM + HUNSEC
      DO WHILE (OTIM > TIM)
        CALL TIMER(TIM)
      END DO
C
      RETURN
      END
